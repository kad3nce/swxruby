                def float_to_bytecode(float) #:nodoc#
                        hex = []
                        [float].pack('E').each_byte { |byte| hex << '%02X' % byte }
                                                                                                                        # Aral did this in SWX PHP, so I'm doing it here
                        DataTypeCodes::FLOAT + (hex[4..-1] + hex[0..3]).join
                end
